import { Link, Spacer, Text, User, useTheme, Grid } from '@geist-ui/react'
import NextLink from 'next/link'
import Sponsors from '../../sponsors'

export const meta = {
  title: '感谢名单',
  date: '2020-04-08T03:20:00.763Z',
}

export const SponsorList = () => {
  const theme = useTheme()
  const list = Sponsors.map(item => ({
    ...item,
    amount: item.amount.toFixed(2),
  })).sort((a, b) => (a.isSponsor ? -1 : 1))
  return (
    <Grid.Container gap={2}>
      {list.map((item, index) => (
        <Grid key={`${item.name}-${index}`} xs={12} sm={8} alignItems="center">
          <User text={item.name} name={item.name} src={item.avatar}>
            <Text span size="12px">
              {item.amount}${' '}
              <Text type="secondary" span>
                {item.isSponsor ? '/ month' : ''}
              </Text>
            </Text>
          </User>
        </Grid>
      ))}
    </Grid.Container>
  )
}
